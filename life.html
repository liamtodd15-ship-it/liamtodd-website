async function loadPosts() { var list = document.getElementById('post-list'); try { var res = await fetch('/posts/life/index.json'); var posts = await res.json(); if (posts.length === 0) { list.innerHTML = '
  • No posts yet. Check back soon.
    • '; return; } list.innerHTML = posts.map(function(post) { var d = new Date(post.date); var dateStr = d.toLocaleDateString('en-GB', { day: 'numeric', month: 'long', year: 'numeric' }); return '
  • ' + '
    ' + dateStr + '
    ' + '
    ' + post.title + '
    ' + '
    ' + post.excerpt + '
    ' + '
    • '; }).join(''); } catch(e) { console.error('Error loading posts:', e); list.innerHTML = '
  • No posts yet. Check back soon.
    • '; } }
